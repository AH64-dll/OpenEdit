# PyAgent for Kdenlive — Whole-Pipeline Cleanup Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite all 7 phases of `pyagent-kdenlive-guide` for clarity and fix all bugs found during the rewrite, with the 19 existing tools' JSON I/O unchanged. ~3 weeks of work, one commit per phase.

**Architecture:** 3-layer Phase 2 (types / IO / ops); split Phase 3 into per-domain `tools/`; consolidate Phase 4 adapter registry; slim Phase 5 (D-Bus) and Phase 6 (render+QC); collapse Phase 7 to one persistent e2e test. Migration happens phase-by-phase as atomic commits with golden-file tests locking tool I/O.

**Tech Stack:** Python 3.11+, lxml, FastAPI, watchfiles, jeepney (D-Bus), pytest; TypeScript via `pi` extension; melt + ffmpeg + ffprobe (system). All dependencies unchanged from current `pyproject.toml` files.

## Global Constraints

These are the spec's project-wide rules. Every task's requirements implicitly include this section.

- **Python 3.11+** only; no `from __future__ import annotations` removed (keep them).
- **No Godot patterns** (this is not Godot; but: no global state, no `print()` in production, no `import *` in production).
- **All errors** raised to the LLM carry a `fix:` hint line.
- **Three error classes**: `BackendError`, `ValidationError`, `NotFoundError`, `CatalogError` — defined ONCE in `phase2_project_engine/errors.py`.
- **File naming**: snake_case, no spaces, no CamelCase. Per `00_START_HERE.md` §"Asset naming".
- **Module size budget**: every production file <300 lines; every test file <400 lines. Split if a file needs to grow.
- **Tests**: ≥90% coverage on phase2; ≥80% on phase3-6; e2e in phase7.
- **Tool I/O**: the 19 tools' JSON output is locked by golden-file tests. ANY drift fails the build.
- **Bug policy** (per spec): every bug found while rewriting MUST be fixed with a regression test before that phase's commit lands. Log to `BUGS_FIXED.md` (one line per bug, with `file:line`).
- **Commit format**: per phase, a single commit with the message `[phase-N][cleanup] rewrite: <short summary>`. Bug-fix commits during a phase go in the SAME phase commit, not separate.
- **Working tree state**: at every commit boundary, `PYTHONPATH=. pytest -q` is green. Never commit red.
- **No new dependencies** added. Use what's in `pyproject.toml` already.
- **No behavior change** for the 19 existing tools' I/O.

# phase2_project_engine/errors.py
"""Error classes for the editor backend.

All errors raised by backend code inherit BackendError. ValidationError
is the only one the LLM should see in normal flow — it always carries
a `fix:` line so the LLM can self-correct.

Hierarchy:
    BackendError
    ├── ValidationError   (bad input — LLM self-corrects)
    ├── NotFoundError     (clip/track not in project)
    └── CatalogError      (effect/transition not in catalog)
"""
from __future__ import annotations


class BackendError(Exception):
    """Base for all backend errors."""


class ValidationError(BackendError):
    """Bad input. The message MUST contain a `fix:` line."""


class NotFoundError(BackendError):
    """Referenced clip/track/effect/transition not in the project."""


class CatalogError(BackendError):
    """Effect/transition id not in Phase 1's catalog."""


def validation_error(msg: str, fix_hint: str | None = None) -> ValidationError:
    """Build a ValidationError. Adds a `fix:` line if not already present.

    >>> e = validation_error("bad", "do good")
    >>> "fix:" in str(e) and "do good" in str(e)
    True
    """
    if "fix:" in msg:
        return ValidationError(msg)
    if fix_hint is None:
        return ValidationError(msg)
    return ValidationError(f"{msg}\nfix: {fix_hint}")


__all__ = [
    "BackendError", "ValidationError", "NotFoundError", "CatalogError",
    "validation_error",
]
```

# phase2_project_engine/catalog.py
"""In-memory snapshot of Phase 1's catalog.json."""
from __future__ import annotations

import json
from dataclasses import dataclass, field
from pathlib import Path


@dataclass
class Catalog:
    effects: list[dict]
    transitions: list[dict]
    generators: list[dict]
    by_id: dict[str, dict] = field(default_factory=dict)

    @classmethod
    def from_json(cls, path: str | Path) -> "Catalog":
        data = json.loads(Path(path).read_text())
        by_id: dict[str, dict] = {}
        for e in data["effects"] + data["transitions"] + data["generators"]:
            kid = e.get("kdenlive_id")
            if kid:
                by_id[kid] = e
        return cls(
            effects=data["effects"],
            transitions=data["transitions"],
            generators=data["generators"],
            by_id=by_id,
        )


__all__ = ["Catalog"]
```

def playlist_duration(pl: etree._Element | None) -> float:
    if pl is None:
        return 0.0
    duration = 0.0
    for child in pl:
        if child.tag == "entry":
            in_val = _tc_to_sec(child.get("in", "00:00:00.000"))
            out_val = _tc_to_sec(child.get("out", "00:00:00.000"))
            duration += max(0.0, out_val - in_val)
        elif child.tag == "blank":
            duration += _tc_to_sec(child.get("length", "00:00:00.000"))
    return duration

"""Bin operations: import_media."""
from __future__ import annotations

from collections.abc import Sequence
from lxml import etree

from ..errors import BackendError
from ..io import ProjectTree
from ..validators import validate_source_path


def import_media(tree: ProjectTree, paths: Sequence[str]) -> list[str]:
    """Add media files to the project bin.

    Producers are added as children of the <mlt> root (NOT inside the
    main_bin playlist) — that's what Kdenlive's actual file format
    requires. Returns the new kdenlive:id of each imported producer.
    """
    new_ids: list[str] = []
    for p in paths:
        abs_path = validate_source_path(p)
        insert_idx = 0
        for idx, child in enumerate(tree.root):
            if child.tag in ("playlist", "tractor", "transition", "filter"):
                insert_idx = idx
                break
        else:
            insert_idx = len(tree.root)
        producer = etree.Element("producer")
        tree.root.insert(insert_idx, producer)
        producer.set("id", f"producer_{len(tree.root) - 1}")
        resource = etree.SubElement(producer, "property")
        resource.set("name", "resource")
        resource.text = str(abs_path)
        tree.ensure_kdenlive_properties_on_producer(producer, str(abs_path))
        kid = next(
            (pp.text for pp in producer.iter("property")
             if pp.get("name") == "kdenlive:id"),
            None,
        )
        if kid is None:
            raise BackendError(
                f"internal: imported {abs_path} has no kdenlive:id"
            )
        new_ids.append(kid)
    return new_ids

def compare_key_subset(actual, expected):
    """Recursively compare, but skip 'id' fields and timestamps."""
    SKIP_KEYS = {"modified", "created", "uuid", "control_uuid"}
    if isinstance(expected, dict):
        assert isinstance(actual, dict)
        for k, v in expected.items():
            if k in SKIP_KEYS:
                continue
            assert k in actual, f"missing key: {k}"
            compare_key_subset(actual[k], v)
    elif isinstance(expected, list):
        assert isinstance(actual, list)
        for i, v in enumerate(expected):
            if i < len(actual):
                compare_key_subset(actual[i], v)
    else:
        assert actual == expected
