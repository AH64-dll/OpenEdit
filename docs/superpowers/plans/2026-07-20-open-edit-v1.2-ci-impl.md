```yaml
# Open Edit v1.2 — CI workflow
#
# Un-ignores the bwrap-dependent tests written in v1.1 T3 (5 Rust
# integration tests + 5 Python e2e tests). Runs on every push to main,
# every PR to main, every tag of the form v*, and manual dispatch.
#
# Three jobs:
#   - python-unit: 308 Python unit tests, matrix on 3.11/3.12/3.14 (~2 min)
#   - rust-build:  cargo build + cargo test on the Rust sandbox (~1 min)
#   - bwrap-tests: un-ignored Rust integration + Python e2e (~5-10 min)
#
# bwrap-tests depends on the other two succeeding (saves CI minutes).
# Concurrency: cancel in-progress on PRs, never cancel on main.

name: CI

on:
  push:
    branches: [main]
    tags: ["v*"]
  pull_request:
    branches: [main]
  workflow_dispatch:

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: ${{ github.event_name == 'pull_request' }}

permissions:
  contents: read

jobs:
  python-unit:
    name: Python unit tests (${{ matrix.python-version }})
    runs-on: ubuntu-latest
    strategy:
      fail-fast: false
      matrix:
        python-version: ["3.11", "3.12", "3.14"]
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python ${{ matrix.python-version }}
        uses: actions/setup-python@v5
        with:
          python-version: ${{ matrix.python-version }}
          cache: pip
          cache-dependency-path: open_edit/pyproject.toml

      - name: Install dependencies
        working-directory: open_edit
        run: |
          python -m pip install --upgrade pip
          pip install -e ".[dev]"

      - name: Run unit tests (excluding bwrap e2e)
        working-directory: open_edit
        run: |
          pytest tests/ --ignore=tests/test_free_form_e2e.py -v

  rust-build:
    name: Rust build + unit tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Rust toolchain
        uses: dtolnay/rust-toolchain@stable

      - name: Cache cargo
        uses: Swatinem/rust-cache@v2
        with:
          workspaces: open_edit/sandbox

      - name: Build
        working-directory: open_edit/sandbox
        run: cargo build --release

      - name: Run unit tests
        working-directory: open_edit/sandbox
        run: cargo test

  bwrap-tests:
    name: bwrap + cgroup tests (un-ignored)
    runs-on: ubuntu-latest
    needs: [python-unit, rust-build]
    steps:
      - uses: actions/checkout@v4

      - name: Install system dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y bubblewrap melt ffmpeg

      - name: Set up Python 3.14
        uses: actions/setup-python@v5
        with:
          python-version: "3.14"
          cache: pip
          cache-dependency-path: open_edit/pyproject.toml

      - name: Set up Rust toolchain
        uses: dtolnay/rust-toolchain@stable

      - name: Cache cargo
        uses: Swatinem/rust-cache@v2
        with:
          workspaces: open_edit/sandbox

      - name: Install Python dependencies
        working-directory: open_edit
        run: |
          python -m pip install --upgrade pip
          pip install -e ".[dev]"

      - name: Setup cgroup v2 for render sandbox
        run: sudo bin/setup_render_cgroup.sh

      - name: Build Rust (release)
        working-directory: open_edit/sandbox
        run: cargo build --release

      - name: Run un-ignored Rust integration tests
        working-directory: open_edit/sandbox
        run: cargo test --features integration -- --include-ignored

      - name: Run un-skipped Python e2e tests
        working-directory: open_edit
        run: pytest tests/test_free_form_e2e.py -v
```
